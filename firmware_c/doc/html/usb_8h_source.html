314 void USB_process_rx_tx(uint32_t epnum){ //only enters this function when last transfer did complete with an ack
315  if(EP_CONFIG_ARRAY[epnum].EP_STATE==EP_IN){ //From device to host PC
316  //We will enter this function BEFORE we have transfered the packet which we will prepare now
317  if(EP_CONFIG_ARRAY[epnum].MultiStage_Bytes!=no_bytes_left){ //bytes to send left
318  if(EP_CONFIG_ARRAY[epnum].MultiStage_Bytes>=USB_SRAM_ENDP_SIZE){ //will be at least one additional transfer
319  for(uint32_t i=0;i<USB_SRAM_ENDP_SIZE;i++){
320  USB_SRAM_EP_ADDR(epnum)[i]=EP_CONFIG_ARRAY[epnum].MultiStage_Storage_ptr[i];
321  }
322  USB_MAXPLD(epnum)=USB_SRAM_ENDP_SIZE;
323  EP_CONFIG_ARRAY[epnum].MultiStage_Storage_ptr+=USB_SRAM_ENDP_SIZE; //shift the storage pointer by endpoint size
324  EP_CONFIG_ARRAY[epnum].MultiStage_Bytes-=USB_SRAM_ENDP_SIZE; //log how many bytes are left
325  //UART0_send_async("um",2,0);
326  }else{ //only this transfer left
327  for(int32_t i=0;i<EP_CONFIG_ARRAY[epnum].MultiStage_Bytes;i++){
328  USB_SRAM_EP_ADDR(epnum)[i]=EP_CONFIG_ARRAY[epnum].MultiStage_Storage_ptr[i];
329  }
330  USB_MAXPLD(epnum)=EP_CONFIG_ARRAY[epnum].MultiStage_Bytes;
331  EP_CONFIG_ARRAY[epnum].MultiStage_Bytes=no_bytes_left; //transfer finished
332 
333  //UART0_send_async("u",1,0);
334  //TODO free storage?
335  }
336  }else{ //now we are actually finished with transmitting, the interrupt says that the last packet has been sent by the ep
337  //free pointer to MultiStage_Storage?
338  EP_CONFIG_ARRAY[epnum].MultiStage_Storage_ptr=NULL;
339  USB_endpoint_finished_last_send(epnum);
340  if(changeAddressTo!=dontChangeAddress){ //must happen after status stage has finished
341  USB_FADDR&=0xffffff80;
342  USB_FADDR|=0x000007f&changeAddressTo; //USB_FADDR overlaps overlaps with usbattr, take care
343  changeAddressTo=dontChangeAddress;
344  }
345  }
346  }else if(EP_CONFIG_ARRAY[epnum].EP_STATE==EP_OUT){ //From host to device
347  //We will enter this AFTER we have recieved data
348  int32_t bytesInBuffer=USB_MAXPLD(epnum); //read how many bytes have been recieved
349  if(bytesInBuffer<=EP_CONFIG_ARRAY[epnum].MultiStage_Bytes){ //we have not yet exeeded the maximum allowed storage space
350  int32_t i=0;
351  while(i<bytesInBuffer){ //read out data from peripheral
352  EP_CONFIG_ARRAY[epnum].MultiStage_Storage_ptr[i]=USB_SRAM_EP_ADDR(epnum)[i];
353  i++;
354  }
355  if(bytesInBuffer==USB_SRAM_ENDP_SIZE){ //there will be another transfer
356  USB_MAXPLD(epnum)=USB_SRAM_ENDP_SIZE; //allow another transfer
357  EP_CONFIG_ARRAY[epnum].MultiStage_Bytes-=bytesInBuffer;
358  EP_CONFIG_ARRAY[epnum].MultiStage_Storage_ptr+=bytesInBuffer;
359  }else{ //last transfer
360  EP_CONFIG_ARRAY[epnum].MultiStage_Bytes=no_bytes_left; //transfer finished
361  EP_CONFIG_ARRAY[epnum].MultiStage_Storage_ptr=NULL;
362  }
363  }
364  }
365 }
